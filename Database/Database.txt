CREATE TABLE Meals
(meal_id INTEGER NOT NULL,
type_meal VARCHAR(30) NOT NULL,
name_meal VARCHAR(30) NOT NULL,
price DECIMAL(5,2) NOT NULL,
CONSTRAINT Number1 PRIMARY KEY(meal_id));

CREATE TABLE Ingredients
(Ing_id INTEGER NOT NULL,
name_ing VARCHAR(30)NOT NULL,
Allergene BOOLEAN,
Meat BOOLEAN,
CONSTRAINT Number2 PRIMARY KEY(Ing_id));

CREATE TABLE Meals_Ing
(meal_id INTEGER NOT NULL,
Ing_id INTEGER NOT NULL,
CONSTRAINT pk_meals_ing PRIMARY KEY(meal_id,Ing_id),
CONSTRAINT meal_id1 FOREIGN KEY(meal_id) REFERENCES Meals(meal_id),
CONSTRAINT Ing_id1 FOREIGN KEY(Ing_id) REFERENCES Ingredients(Ing_id));
